An attempt to reduce repetition from certain commands (like fortunes). Needs testing on specific sample set/size, this

    #!/bin/bash

    while ((1==1))

    do
        until var="$(./seen "$(fortune -s cbbl)" 2>/dev/null)"; do : ; done && echo "$var"

        #sleep 0.1
        
    done

## Gpu utilization indicator in tint2

In tint2rc

    execp = new
    execp_centered = 1
    execp_has_icon = 0
    execp_command = gpu
    execp_continuous = 0
    execp_interval = 1
    execp_font = cuprum 12
    execp_font_color = #111111 70
    execp_padding = 0 0 0
    execp_tooltip = gpu

~/bin/gpu script

    nvidia-smi -q -d UTILIZATION | grep Gpu | grep -Eo "[0-9]{1,3}"
